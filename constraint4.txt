
//  Write a constraint to generate a random even number between 50 and 100.
class first; 
  rand int arr; 
  constraint c1 {
                 arr inside {[50:100]};
                
                } 
  constraint c2 {
   
    arr%2==0;
       
   
  }
endclass

//  . Write a constraint on a two-dimensional array to generate even numbers in the first 
//4 locations and odd numbers in the next 4 locations. 
class first; 
  rand bit [4:0] arr[2][4]; 
  
   constraint c2 {
        foreach(arr[1][i]){
          arr[1][i]%2==1;
        
    }} 
 
  
  
  constraint c1 {
    foreach(arr[0][i]){
      arr[0][i]%2==0;
        
   } } 
     
 
endclass
