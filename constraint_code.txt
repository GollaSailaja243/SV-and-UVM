//Write a constraint to generate a variable with 0-31 bits should be 1, 
32-61 bits should be 0. 
class seq_item;
  rand bit [0:61] val;
 
  constraint ca{
    foreach(val[i]){
      if(i<32)
        val[i]==1;
      
      else
        val[i]==0;
    
    }
  }
    
endclass

// Write a constraint to generate FACTORIAL of first 5 even numbers and odd numbers .

class seq_item;
  rand int even[];
  rand int odd[];
  constraint arra_a{
    
    even.size()==5;
    odd.size==5;
   
  }
  
  constraint ca{
    foreach(even[i]){
      even[i]==fact(2*i);
      
     
    }}
    
      constraint odder{ 
        foreach(odd[i]){
          odd[i]==fact((2*i)+1);}
    
      }
  function int fact(int a);
    if(a==0)
      return 1;
    else
     return a*fact(a-1);
    
     endfunction
    
    
    
endclass



module constraint_example;

  initial begin
    item = new();
    repeat(1)begin
    item.randomize();
      $display("%p",item.even);
      $display("%p",item.odd);
    end
  end
endmodule
